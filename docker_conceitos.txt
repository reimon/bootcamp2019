Dockerfile
    Receita para criar uma imagem


# Partimos de uma imagem existente
FROM node:10

# Definimos a pasta e copiamos os arquivos
WORKDIR /usr/app
COPY . ./

# Instalamos as dependencias
RUN yarn

# Qual porta queremos expor?
EXPOSE 3333

# Executamos nossa aplicação
CMD yarn start
